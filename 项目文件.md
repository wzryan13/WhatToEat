```python
class UserProfile(BaseModel):
    schema_version: int = 1

    # ── 硬约束(不可违反, 硬筛选阶段排除) ──
    allergies: list[MemoryFact] = []              # 花生、海鲜
    food_blacklist: list[MemoryFact] = []         # 明确不吃的食材
    religious_restrictions: list[MemoryFact] = []  # 清真、素食
    disliked_cuisines: list[MemoryFact] = []       # 明确不吃的菜系(不吃烧烤)

    # ── 软偏好(影响 rerank 排序) ──
    spice_tolerance: MemoryFact | None = None     # 不辣/微辣/中辣/重辣
    sweetness: MemoryFact | None = None           # 不甜/微甜/嗜甜
    cuisine_tags: dict[str, str] = {}             # {"日料": "liked", "川菜": "loved"}
    health_goals: list[MemoryFact] = []           # 低碳、高蛋白、减脂

    # ── 消费信息 ──
    home_area: MemoryFact | None = None           # 常驻区域
    budget_solo: MemoryFact | None = None         # 一人食人均
    budget_group: MemoryFact | None = None        # 聚餐人均
    active_areas: list[MemoryFact] = []           # 活动区域
    default_city: MemoryFact | None = None        # 默认城市

    # ── 行为偏好(后期实现) ──
    # exploration: MemoryFact | None = None       # 探索度
    # decision_factors: list[str] = []            # 决策因子排序
```

### MemoryFact 结构

```python
class MemoryFact(BaseModel):
    value: str
    source: str = ""           # "用户声明" / "对话推断" / "行为累积"
    confidence: float = 0.5    # 0~1, 当前阶段硬编码, 后续由 LLM③ 给出
    updated_at: str = ""
    last_seen_at: str = ""
```

### cuisine_tags 标签规则

```
用户第一次提到喜欢某菜系 → "liked"
用户多次提到或强烈表达 → "loved"

使用方式:
  - "liked": rerank 时加权
  - "loved": rerank 时强加权 + 兜底搜索时优先用
```

```
  - 本轮 negative_conditions
```

```python
def build_memory_for_intent(profile: UserProfile, session: SessionMemory) -> str:
    """城市 + 硬约束 + session 连续性"""
    lines = []
    if profile.default_city:
        lines.append(f"用户默认城市: {profile.default_city.value}")
    if profile.allergies:
        lines.append("过敏: " + "、".join(f.value for f in profile.allergies))
    if profile.food_blacklist:
        lines.append("不吃的食材: " + "、".join(f.value for f in profile.food_blacklist))
    if profile.religious_restrictions:
        lines.append("饮食限制: " + "、".join(f.value for f in profile.religious_restrictions))
    if profile.disliked_cuisines:
        lines.append("不吃的菜系: " + "、".join(f.value for f in profile.disliked_cuisines))
    if session.active_city:
        lines.append(f"当前城市: {session.active_city}")
    if session.active_location_text:
        lines.append(f"当前位置: {session.active_location_text}")
    if session.active_budget_range:
        budget_parts = []
        if session.active_budget_range.min is not None:
            budget_parts.append(f"最低{session.active_budget_range.min}元")
        if session.active_budget_range.max is not None:
            budget_parts.append(f"最高{session.active_budget_range.max}元")
        if budget_parts:
            lines.append("当前预算: " + ", ".join(budget_parts))
    return "\n".join(lines) if lines else "暂无记忆信息。"


def build_memory_for_rerank(profile: UserProfile, session: SessionMemory) -> str:
    """偏好 + 限制 + 口味 + 预算"""
    lines = []
    if profile.allergies:
        lines.append("过敏: " + "、".join(f.value for f in profile.allergies))
    if profile.food_blacklist:
        lines.append("不吃: " + "、".join(f.value for f in profile.food_blacklist))
    if profile.disliked_cuisines:
        lines.append("不吃的菜系: " + "、".join(f.value for f in profile.disliked_cuisines))
    if profile.cuisine_tags:
        liked = [k for k, v in profile.cuisine_tags.items() if v == "liked"]
        loved = [k for k, v in profile.cuisine_tags.items() if v == "loved"]
        if loved:
            lines.append("最爱菜系: " + "、".join(loved))
        if liked:
            lines.append("喜欢菜系: " + "、".join(liked))
    if profile.spice_tolerance:
        lines.append(f"辣度接受: {profile.spice_tolerance.value}")
    if profile.sweetness:
        lines.append(f"甜度偏好: {profile.sweetness.value}")
    if profile.health_goals:
        lines.append("健康目标: " + "、".join(f.value for f in profile.health_goals))
    if profile.budget_solo:
        lines.append(f"一人食预算: {profile.budget_solo.value}")
    if profile.budget_group:
        lines.append(f"聚餐预算: {profile.budget_group.value}")
    if session.active_negative_conditions:
        lines.append("本轮排除: " + "、".join(session.active_negative_conditions))
    return "\n".join(lines) if lines else "暂无用户偏好。"
```

```
{old_profile}
```

## 六、memory/store.py 改造

### 新增方法: apply_profile_updates

```python
async def apply_profile_updates(self, user_id: str, updates: list[ProfileUpdate]) -> None:
    """根据 LLM③ 的指令批量更新画像字段"""
    profile = await self._load_profile(user_id)

    for update in updates:
        field_parts = update.field.split(".")

        if field_parts[0] == "cuisine_tags" and len(field_parts) == 2:
            cuisine = field_parts[1]
            if update.action == "add":
                profile.cuisine_tags[cuisine] = update.tag_level or "liked"
            elif update.action == "remove":
                profile.cuisine_tags.pop(cuisine, None)

        elif field_parts[0] in ("allergies", "food_blacklist", "religious_restrictions",
                                 "disliked_cuisines", "health_goals", "active_areas"):
            fact_list = getattr(profile, field_parts[0])
            if update.action == "add":
                if not any(f.value == update.value for f in fact_list):
                    fact_list.append(MemoryFact(
                        value=update.value,
                        source="LLM推断",
                        confidence=0.8,
                        updated_at=now_iso(),
                        last_seen_at=now_iso(),
                    ))
            elif update.action == "remove":
                setattr(profile, field_parts[0],
                        [f for f in fact_list if f.value != update.value])

        elif field_parts[0] in ("spice_tolerance", "sweetness", "home_area",
                                 "budget_solo", "budget_group", "default_city"):
            if update.action == "set":
                setattr(profile, field_parts[0], MemoryFact(
                    value=update.value,
                    source="LLM推断",
                    confidence=0.8,
                    updated_at=now_iso(),
                    last_seen_at=now_iso(),
                ))
            elif update.action == "remove":
                setattr(profile, field_parts[0], None)

    await self._save_profile(user_id, profile)
```
